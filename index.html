ระบบบันทึกข้อมูลเข้า Google Sheets โดยตรงแบบอัตโนมัติ (Real-time Auto-Sync)

ตั้งค่าเพียงครั้งเดียว ระบบจะทำการส่งข้อมูลการประเมินของพยาบาลทุกคนเข้าสู่ Google Sheets ของคุณทันทีที่กดยืนยันบันทึกข้อมูล โดยข้อมูลจะแชร์ร่วมกันและอัปเดตแบบเรียลไทม์

ตั้งค่าการเชื่อมต่อ Google Sheets

วาง URL ที่ได้จากขั้นตอน "Deploy as Web App" ใน Google Sheets ของคุณ

การจัดการข้อมูลย้อนหลัง

หากเคยประเมินไปแล้วก่อนตั้งค่าเชื่อมโยงชีต คุณสามารถกดส่งประวัติทั้งหมดเข้าชีตได้ทันทีในคลิกเดียว

ทางเลือกสำรอง (ส่งออกข้อมูลแบบเดิม)
คัดลอกเรียบร้อย! วาง (Ctrl+V) บนหน้าชีตได้เลย

วิธีการเชื่อมโยง Google Sheets ภายใน 1 นาที

1

เปิดไฟล์ Google Sheets ของคุณ

เปิดหน้า Google Sheets ว่างๆ ขึ้นมา 1 หน้า จากนั้นไปที่เมนู **ส่วนขยาย (Extensions)** > **Apps Script**

2

คัดลอกโค้ดสคริปต์ไปวาง

ลบโค้ดเดิมในโปรแกรมแก้ไขสคริปต์ออกให้หมด แล้วกดคัดลอกกล่องโค้ดสีขาวด้านล่างนี้ไปวางแทนที่ทั้งหมด

function doPost(e) {
  try {
    var jsonString = e.postData.contents;
    var data = JSON.parse(jsonString);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    // สร้างหัวตารางกรณีชีตยังว่างเปล่าอยู่
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        "รหัสบันทึก", "ชื่อ", "นามสกุล", "หน่วยงานสังกัด", "วันที่เริ่มทำงาน", 
        "คะแนนเฉลี่ยร้อยละ", "ข้อเสนอแนะหัวหน้างาน", "ความเห็นเพิ่มเติม", 
        "ผู้ประเมิน", "ตำแหน่งผู้ประเมิน", "วันเวลาประเมิน"
      ]);
    }
    
    // เพิ่มแถวข้อมูลใหม่
    sheet.appendRow([
      data.id || "",
      data.nurseFirstname || "",
      data.nurseLastname || "",
      data.department || "",
      data.startDate || "",
      data.weightedScore || 0,
      data.decision || "",
      data.comments || "",
      data.assessorName || "",
      data.assessorRole || "",
      data.evaluatedAt || ""
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({"status": "success"}))
                         .setMimeType(ContentService.MimeType.JSON)
                         .setHeader("Access-Control-Allow-Origin", "*");
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()}))
                         .setMimeType(ContentService.MimeType.JSON)
                         .setHeader("Access-Control-Allow-Origin", "*");
  }
}
3

บันทึกและทำการสร้าง Deployment เป็น Web App

กดปุ่ม **บันทึก (Save)** (รูปแผ่นดิสก์) จากนั้นเลือกเมนู **การใช้งานได้จริง (Deploy)** > **การนำไปใช้งานได้จริงใหม่ (New deployment)**

4

ตั้งค่าสิทธิ์ให้เป็นสาธารณะเพื่อให้ส่งข้อมูลได้

1. เลือกประเภทเป็น **แอปพลิเคชันเว็บ (Web app)**
2. ตั้งค่าการเข้าถึง (Who has access) ให้เลือกเป็น **ทุกคน (Anyone)** (เพื่อให้แอปนี้ยิง API บันทึกข้อมูลเข้าให้ได้)
3. กดปุ่ม **การนำไปใช้งานได้จริง (Deploy)** และอนุมัติสิทธิ์ (Authorize Access)
4. คัดลอก **URL แอปพลิเคชันเว็บ (Web App URL)** มาวางลงในช่องทางด้านซ้ายมือ แล้วกดบันทึกได้เลย!
